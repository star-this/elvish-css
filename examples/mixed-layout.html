Balanced proportions for technical documentation

fluid-typography.css 
/* Fluid: scales with viewport */
h1 { font-size: var(--step-5); }
h2 { font-size: var(--step-4); }
body { font-size: var(--step-0); }
small { font-size: var(--step--1); }

/* Result: 18px ‚Üí 20px base,
   scaling proportionally */
space-pairs.css 
/* Space Pair: two sizes, one variable */
.hero {
  padding: var(--space-xl-3xl);
  /* 54px on mobile ‚Üí 120px on desktop */
}

.card-grid {
  gap: var(--space-s-l);
  /* 18px tight ‚Üí 40px spacious */
}
reverse-pairs.css 
/* Reverse: SHRINKS as viewport grows */
.quote {
  padding: var(--space-2xl-l);
  /* 72px mobile ‚Üí 40px desktop */
  /* Mobile needs breathing room! */
}

.nav-links {
  gap: var(--space-l-s);
  /* Touch-friendly ‚Üí mouse-friendly */
}
static-ratios.css 
/* Static: same size at ALL viewports */
.pricing {
  --ratio: var(--ratio-silver);
  padding: var(--s2);
  /* Always 2rem (silver) - dense, readable */
}

.hero {
  --ratio: var(--ratio-golden);
  /* Dramatic proportions */
}
